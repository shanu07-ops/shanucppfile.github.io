11. Convert Binary to Decimal

Source Code:


#include 
#include 
#include 
using namespace std;

int main() {
    string binary = "1010";
    int decimal = 0;
    for (int i = 0; i < binary.length(); i++) {
        if (binary[i] == '1') {
            decimal += pow(2, binary.length() - 1 - i);
        }
    }
    cout << "Decimal: " << decimal << endl;
    return 0;
}

Output:

Decimal: 10

12. Convert Decimal to Binary

Source Code:


#include 
#include 
using namespace std;

int main() {
    int decimal = 13;
    string binary = "";
    while (decimal > 0) {
        binary = to_string(decimal % 2) + binary;
        decimal /= 2;
    }
    cout << "Binary: " << binary << endl;
    return 0;
}

Output:

Binary: 1101
